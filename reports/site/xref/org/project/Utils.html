36      /**
37       * Inverts a given map by exchanging each entry's key and value.
38       *
39       * @param map source map
40       * @return inverted map
41       */
42      private static TreeMap<Character, Integer> invertMap(SortedMap<Integer, String> map) {
43          TreeMap<Character, Integer> newMap = new TreeMap<>();
44          for (Map.Entry<Integer, String> entry : map.entrySet()) {
45              if (entry.getValue().length() < 2) {
46                  newMap.put(entry.getValue().charAt(0), entry.getKey());
47              }
48          }
49          return newMap;
50      }

52      /**
53       * Checks given string to make sure that it contains
54       * valid roman numerals.
55       *
56       * @param input string to check
57       * @return true if string contains valid chars
58       */
59      public static boolean containsValidChars(String input) {
60          boolean result = true;
61          for (int i = 0; i < input.length(); i++) {
62              if (!ROMAN_TO_INTEGER.containsKey(input.charAt(i))) {
63                  result = false;
64                  break;
65              }
66          }
67          return result;
68      }
